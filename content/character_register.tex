\newglossaryentry{Diemut von Raabs}{
  type=characters,
  name={Diemut von Raabs},
  description={Niedere Adelige, Tochter \gls{Wolfhart von Raabs}s. Einstige Verlobte \gls{Kuonrat von Steinare}s. Der Vorname ist regional belegt (vgl. \newglossaryentry{Kuonrat von Hardegg}{
  type=characters,
  name={Kuonr\^{a}t von Hardegg},

\newglossaryentry{Einsidel}{
